🔍 程式碼解析 1 — 風格物件 STYLES

這段的目的：用一個物件儲存 6 組配色方案，切換風格只要換一組參數。

var STYLES = {
  1: {
    name: '📘 專業商務',
    coverBg: '#1a237e',     // 封面背景色
    headerBg: '#283593',    // 內容頁標題橫條
    titleColor: '#ffffff',  // 標題文字色
    subtitleColor: '#90caf9', // 副標題文字色
    contentBg: '#f5f5f5',   // 內容頁背景
    bulletColor: '#333333', // 重點文字色
    accentBar: '#1565c0',   // 左側裝飾條
    endBg: '#1a237e'        // 結尾頁背景
  },
  2: { name: '🌿 清新自然', ... },
  3: { name: '🚀 科技未來', ... },
  // ... 共 6 種
};
  • STYLES — 是一個物件，key 是 1~6 的數字，value 是每種風格的配色參數
  • STYLES[1] — 取得第 1 種風格（專業商務）的所有配色
  • style.coverBg — 取得該風格的封面背景色
  • 要新增第 7 種風格？只要在 STYLES 裡加一組 key-value 就好！
💡 為什麼用物件？把配色抽出來當參數，所有的 createCoverSlide()、createContentSlides() 函式都接收 style 物件，根據參數來設定顏色。這樣同一份程式碼就能產出 6 種完全不同風格的簡報！

🔍 程式碼解析 2 — 讀取試算表資料

這段的目的：從試算表中讀取使用者填寫的主題、關鍵字和風格選擇。

function getInputFromSheet() {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheet = ss.getActiveSheet();

  // A1 = 主題
  var topic = sheet.getRange('A1').getValue().toString().trim();

  // B 欄 = 關鍵字列表
  var keywords = [];
  var lastRow = sheet.getLastRow();
  for (var i = 1; i <= lastRow; i++) {
    var val = sheet.getRange('B' + i).getValue().toString().trim();
    if (val) keywords.push(val);
  }

  // C1 = 風格編號（1~6）
  var styleNum = parseInt(sheet.getRange('C1').getValue()) || 1;
  if (styleNum < 1 || styleNum > 6) styleNum = 1;

  return { topic: topic, keywords: keywords, styleNum: styleNum };
}
  • getRange('A1').getValue() — 取得 A1 儲存格的值
  • toString().trim() — 轉成字串並去除前後空格
  • for 迴圈 — 從 B1 開始往下讀，有值就加入 keywords 陣列
  • parseInt() || 1 — 轉成數字，如果無法轉換就用預設值 1
  • return { ... } — 回傳一個包含三個欄位的物件

🔍 程式碼解析 3 — Prompt 設計（含關鍵字）

這段的目的：把主題和關鍵字組合成 Prompt，要求 AI 回傳 JSON 格式的大綱。

function buildPrompt(topic, keywords) {
  var prompt = '請幫我製作一份關於「' + topic + '」的簡報大綱。\n\n';

  if (keywords.length > 0) {
    prompt += '請特別包含以下重點或關鍵字：\n';
    for (var i = 0; i < keywords.length; i++) {
      prompt += '• ' + keywords[i] + '\n';
    }
    prompt += '\n';
  }

  prompt += '回覆格式必須是純 JSON（不要加 ```json 標記），結構如下：\n';
  prompt += '{ "title": "...", "subtitle": "...", '
         +  '"slides": [{ "heading": "...", "bullets": [...] }], '
         +  '"conclusion": "..." }\n\n';
  prompt += '請產生 5~8 頁，每頁 3~4 個重點，每個重點 25 字以內。';

  return prompt;
}
  • if (keywords.length > 0) — 如果有填關鍵字，才加入 Prompt；沒填也不影響
  • 回傳 JSON 格式 — 明確告訴 AI 要回傳什麼結構，程式才能解析
  • 限制字數 — 「每個重點 25 字以內」避免文字太多溢出投影片
💡 關鍵字的作用：如果你只填主題「人工智慧」，AI 會自己決定要講什麼。但如果你在 B 欄填了「機器學習、教育應用、未來趨勢」，AI 就會特別在簡報中包含這些內容。關鍵字 = 你引導 AI 的方向！

🔍 程式碼解析 4 — 主函式 generatePresentation()

這段的目的：串起所有函式，完成「讀取 → AI → 解析 → 建立 → 記錄」的完整流程。

function generatePresentation(styleOverride) {
  var input = getInputFromSheet();        // 讀取試算表
  var style = STYLES[styleOverride || input.styleNum] || STYLES[1];

  var prompt = buildPrompt(input.topic, input.keywords);  // 組 Prompt
  var aiReply = callGemini(prompt);       // 呼叫 AI
  aiReply = aiReply.replace(/```json\n?/g, '')            // 清理格式
                   .replace(/```\n?/g, '').trim();
  var outline = JSON.parse(aiReply);      // 解析 JSON

  var pres = SlidesApp.create(outline.title);  // 建立簡報
  createCoverSlide(pres, outline.title, outline.subtitle, style);
  createContentSlides(pres, outline.slides, style);
  createEndSlide(pres, outline.conclusion, style);

  return { success: true, url: pres.getUrl(), ... };
}

  • withFailureHandler — 函式失敗後執行的回呼
💡 HtmlService 的好處：讓你在 GAS 中建立真正的網頁 UI！側邊欄可以有按鈕、表單、CSS 樣式，使用者操作起來就像在用一個小工具，完全不需要看程式碼。

❓ 常見問題排解

🔍 怎麼找到錯誤訊息？

  1. 在 Apps Script 編輯器左側選單，找到「⏱ 執行項目」圖示（時鐘圖示），點擊它
  2. 找到最近一筆執行記錄，如果狀態是紅色「失敗」，點擊展開看詳細資訊

❌ 403 或 429 錯誤

原因：API Key 無效或超過免費額度

  • 確認 API Key 沒有多餘空格
  • 免費版每分鐘限 15 次請求
  • 等一分鐘再試

❌ JSON 解析失敗

原因：AI 回覆格式不正確

  • 再按一次「生成」（AI 每次回覆不同）
  • 通常第二次就會成功
  • 程式碼已內建清除 markdown 標記

❌ 選單沒出現

原因：onOpen 尚未執行

  • 手動執行一次 onOpen 函式
  • 回到試算表按 F5 重新整理
  • 等 3~5 秒讓頁面載入

❌ 簡報排版很醜

原因：AI 生成的文字太多

  • Prompt 已限制每點 25 字以內
  • 可在 B 欄加更具體的關鍵字引導
  • 生成後可手動微調簡報內容
