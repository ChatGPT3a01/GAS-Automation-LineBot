Step 2：Webhook 入口

LINE 送來的訊息都會進入 doPost，這是整個系統的大門

/** LINE Webhook 接收點 */
function doPost(e) {
  try {
    const events = JSON.parse(e.postData.contents).events;

    for (let i = 0; i < events.length; i++) {
      handleEvent(events[i]);  // 逐一處理每個事件
    }

    return ContentService.createTextOutput(
      JSON.stringify({ status: 'ok' })
    ).setMimeType(ContentService.MimeType.JSON);
  } catch (error) {
    console.error('doPost 錯誤:', error);
    return ContentService.createTextOutput(
      JSON.stringify({ status: 'error' })
    ).setMimeType(ContentService.MimeType.JSON);
  }
}

/** Webhook 驗證用 */
function doGet(e) {
  return ContentService.createTextOutput('自動化評量系統運作中！');
}
重點： doPost 是 LINE 送訊息進來的入口，doGet 則用來驗證 Webhook 是否活著。
這個結構跟 Part 1、2 完全一樣！

// 使用者狀態管理 — 用 ScriptProperties 存儲
const scriptProps = PropertiesService.getScriptProperties();

function setUserState(userId, state) {
  scriptProps.setProperty('state_' + userId, state);
}

function getUserState(userId) {
  return scriptProps.getProperty('state_' + userId) || '';
}

function clearUserState(userId) {
  scriptProps.deleteProperty('state_' + userId);
}
小知識：ScriptProperties 是 GAS 內建的「鍵值對」存儲，資料會永久保存直到你刪除它，很適合存放簡單的狀態資訊。

Step 5：下載檔案 & 存入 Drive

從 LINE 伺服器下載使用者上傳的 PDF，存到 Google Drive

從 LINE 下載檔案

function downloadFromLine(messageId) {
  const url =
    'https://api-data.line.me/v2/bot/message/'
    + messageId + '/content';

  const res = UrlFetchApp.fetch(url, {
    headers: {
      'Authorization': 'Bearer ' + LINE_TOKEN
    },
    muteHttpExceptions: true
  });

  if (res.getResponseCode() !== 200) {
    throw new Error('檔案下載失敗');
  }
  return res.getBlob();
}

存入 Google Drive

function saveToDrive(blob, fileName, userId) {
  const folder =
    DriveApp.getFolderById(FOLDER_ID);

  // 加上時間戳避免重名
  const timestamp =
    Utilities.formatDate(
      new Date(), 'Asia/Taipei',
      'yyyyMMdd_HHmmss'
    );
  const newName = timestamp + '_'
    + userId.substring(0, 8)
    + '_' + fileName;

  blob.setName(newName);
  return folder.createFile(blob);
}
重點： LINE 的檔案 API 網址是 api-data.line.me（不是一般的 api.line.me）。
檔案名稱加上時間戳和使用者 ID 前 8 碼，方便管理不會重複。

Step 6：PDF 文字提取（關鍵步驟）

利用 Drive API 將 PDF 轉換為 Google Docs，就能讀出文字！

/**
 * PDF → 文字提取
 * 原理：Drive API 可以把 PDF「轉換」成 Google Docs
 *       轉換後的 Docs 就包含了 PDF 裡的文字
 * ⚠️ 必須啟用 Drive API 進階服務！
 */
function extractPdfText(file) {
  try {
    const blob = file.getBlob();
    const tempName = 'temp_pdf_' + Date.now();

    // 1. 將 PDF 轉換為 Google Docs
    const docFile = Drive.Files.insert(
      { title: tempName,
        mimeType: 'application/vnd.google-apps.document' },
      blob,
      { convert: true }
    );

    // 2. 讀取轉換後的文字
    const doc = DocumentApp.openById(docFile.id);
    const text = doc.getBody().getText();

    // 3. 清除暫存檔
    DriveApp.getFileById(docFile.id).setTrashed(true);

    return text;
  } catch (error) {
    console.error('PDF 提取失敗：' + error.message);
    return '';
  }
}
⚠️ 常見錯誤：如果出現 Drive is not defined，代表你忘了啟用 Drive API！
→ GAS 編輯器 → 左側「服務 +」→ 搜尋 Drive API → 新增
限制：這個方法只能提取「文字型 PDF」。如果是掃描檔（圖片型），會提取不到文字。

呼叫 Gemini API（推薦）

function callGemini(systemPrompt, userPrompt) {
  const url = 'https://generativelanguage.googleapis.com/v1beta/models/'
    + GEMINI_MODEL + ':generateContent?key=' + AI_API_KEY;

  const payload = {
    system_instruction: { parts: [{ text: systemPrompt }] },
    contents: [{ parts: [{ text: userPrompt }] }],
    generationConfig: { temperature: 0.7, maxOutputTokens: 500 }
  };

  const res = UrlFetchApp.fetch(url, {
    method: 'post', contentType: 'application/json',
    payload: JSON.stringify(payload), muteHttpExceptions: true
  });
  const result = JSON.parse(res.getContentText());

  if (result.error) throw new Error('Gemini 錯誤：' + result.error.message);
  return result.candidates[0].content.parts[0].text.trim();
}
Prompt 設計技巧：指定「字數」、「語氣」、「結構」三要素，AI 生成的品質會大幅提升！

Step 8：記錄到試算表

每筆評語都自動記錄到 Google Sheets，方便老師追蹤管理

function logToSheet(userId, fileName, feedback) {
  try {
    const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
    let sheet = ss.getSheetByName('評語記錄');

    // 若工作表不存在，自動建立並加標題
    if (!sheet) {
      sheet = ss.insertSheet('評語記錄');
      sheet.getRange(1, 1, 1, 5).setValues([
        ['時間', '使用者 ID', '檔案名稱', '評語', '狀態']
      ]);
      sheet.getRange(1, 1, 1, 5).setFontWeight('bold');
    }

    // 新增一筆記錄
    const timestamp = Utilities.formatDate(
      new Date(), 'Asia/Taipei', 'yyyy/MM/dd HH:mm:ss'
    );
    const maskedId = userId.substring(0, 10) + '***';

    sheet.appendRow([timestamp, maskedId, fileName, feedback, '完成']);
  } catch (error) {
    console.error('試算表記錄錯誤:', error);
  }
}

Step 9：核心處理流程

把所有步驟串起來！這個函式是整個系統的心臟

function processFileAsync(userId, messageId, fileName) {
  try {
    // Step 1：從 LINE 下載檔案
    const fileBlob = downloadFromLine(messageId);

    // Step 2：存到 Google Drive
    const driveFile = saveToDrive(fileBlob, fileName, userId);

    // Step 3：提取 PDF 文字
    const pdfText = extractPdfText(driveFile);
    if (!pdfText || pdfText.trim().length < 10) {
      pushText(userId, '⚠️ 無法讀取 PDF 內容');
      return;
    }

    // Step 4：AI 生成評語
    const feedback = generateFeedback(pdfText);

    // Step 5：記錄到試算表
    logToSheet(userId, fileName, feedback);

    // Step 6：回傳評語給學生
    pushText(userId,
      '✨ 你的自主學習評語 ✨\n\n' + feedback
    );

  } catch (error) {
    console.error('processFileAsync 錯誤:', error);
    pushText(userId, '❌ 處理時發生錯誤：' + error.message);
  }
}
為什麼用 pushText 不用 replyText？
因為 replyToken 只有 30 秒有效期。PDF 處理可能超過 30 秒，
所以先用 replyText 回覆「處理中」，處理完再用 pushText 主動推送結果。
