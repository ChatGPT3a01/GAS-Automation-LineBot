🚪

雙入口系統

同一個 GAS 專案同時作為 Web App（doGet）和 Line Bot Webhook（doPost），共用同一個 searchData() 查詢函式

💡 為什麼要做雙入口？
有些人喜歡在電腦上開網頁查詢，有些人習慣用 Line 傳訊息查。雙入口讓不同習慣的使用者都能用自己最順手的方式查資料！

🔍 程式碼解析 1：doGet() — Web App 入口

這段的目的：當使用者用瀏覽器開啟 Web App 網址時，顯示 HTML 查詢介面；若帶有查詢參數，則直接回傳 JSON。

function doGet(e) {
  // 如果有查詢參數，回傳 JSON（API 模式）
  if (e && e.parameter && e.parameter.action === 'search') {
    var keyword = e.parameter.q || '';
    var results = searchData(keyword);
    return ContentService.createTextOutput(JSON.stringify({
      success: true,
      count: results.length,
      data: results
    })).setMimeType(ContentService.MimeType.JSON);
  }

  // 否則顯示 HTML 查詢頁面
  return HtmlService.createHtmlOutputFromFile('查詢介面')
    .setTitle('資料查詢系統')
    .setXFrameOptionsMode(HtmlService.XFrameOptionsMode.ALLOWALL);
}

逐行解釋：

  • doGet(e) — GAS 的特殊函式，當有人用瀏覽器（GET 請求）開啟 Web App URL 時自動執行，e 包含網址的查詢參數
  • e.parameter.action === 'search' — 檢查網址是否帶有 ?action=search&q=關鍵字 參數（API 模式用）
  • ContentService.createTextOutput() — 回傳純文字（這裡是 JSON 格式），讓其他程式可以串接
  • HtmlService.createHtmlOutputFromFile('查詢介面') — 讀取專案中名為「查詢介面」的 HTML 檔案，作為 Web App 頁面回傳給瀏覽器
  • .setTitle('資料查詢系統') — 設定瀏覽器分頁標題
  • .setXFrameOptionsMode(ALLOWALL) — 允許 HTML 頁面被嵌入 iframe 中（預設會被禁止）
💡 雙模式設計的好處
  • 網頁模式（無參數）：直接開啟 URL → 看到漂亮的查詢介面，適合人類使用者

🔍 程式碼解析 2：searchData() — 模糊搜尋邏輯

這段的目的：這是整個系統的核心，從試算表中搜尋所有包含關鍵字的資料列。

function searchData(keyword) {
  var ss = SpreadsheetApp.openById(SPREADSHEET_ID);
  var sheet = ss.getSheetByName(SHEET_NAME);

  if (!sheet) {
    Logger.log('找不到工作表：' + SHEET_NAME);
    return [];
  }

  var data = sheet.getDataRange().getValues();
  if (data.length < 2) return [];  // 只有標題列或空的

  var headers = data[0];  // 第一列為標題
  var results = [];
  var searchKey = keyword.toLowerCase();

  // 從第二列開始搜尋
  for (var i = 1; i < data.length; i++) {
    var row = data[i];
    var match = false;

    // 檢查每一欄是否包含關鍵字
    for (var j = 0; j < row.length; j++) {
      var cellValue = String(row[j]).toLowerCase();
      if (cellValue.indexOf(searchKey) !== -1) {
        match = true;
        break;
      }
    }

    if (match) {
      var record = {};
      for (var k = 0; k < headers.length; k++) {
        record[headers[k]] = row[k];
      }
      results.push(record);
    }
  }
  return results;
}

逐行解釋：

  • SpreadsheetApp.openById(SPREADSHEET_ID) — 用 ID 開啟試算表
  • ss.getSheetByName(SHEET_NAME) — 找到名為「人員資料」的工作表
  • sheet.getDataRange().getValues() — 一次讀取整張工作表的所有資料，回傳二維陣列
  • data[0] — 第一列是標題（姓名、部門、職稱...）
  • keyword.toLowerCase() — 把關鍵字轉小寫，實現不分大小寫搜尋
  • String(row[j]).toLowerCase() — 把每個欄位值也轉小寫後比對
  • cellValue.indexOf(searchKey) !== -1 — 如果欄位值包含關鍵字，indexOf 會回傳位置（>= 0），不包含則回傳 -1
  • record[headers[k]] = row[k] — 把命中的資料列組成物件（例如 {姓名: "王小明", 部門: "業務部", ...}）
⚠️ 效能提醒
此搜尋方式適合小型資料庫（幾百到幾千筆）。getDataRange().getValues() 會一次把所有資料讀入記憶體，資料量太大會變慢。

🔍 程式碼解析 3：doPost() — Line Bot Webhook

這段的目的：接收 Line Bot 的 Webhook 事件，判斷使用者是否輸入查詢指令（以「查」或「找」開頭），呼叫 searchData() 後回覆結果。

function doPost(e) {
  try {
    var data = JSON.parse(e.postData.contents);
    var events = data.events;

    for (var i = 0; i < events.length; i++) {
      var event = events[i];

      if (event.type === 'message' && event.message.type === 'text') {
        var userMessage = event.message.text.trim();
        var replyToken = event.replyToken;

        // 判斷是否為查詢指令
        if (userMessage.indexOf('查') === 0 || userMessage.indexOf('找') === 0) {
          var keyword = userMessage.substring(1).trim();
          handleLineSearch(replyToken, keyword);
        } else if (userMessage === '使用說明' || userMessage === '幫助') {
          replyText(replyToken, '📖 查詢系統使用說明\n\n' +
            '傳送以下格式即可查詢：\n• 查 王小明\n• 找 業務部\n• 查 test@email\n\n' +
            '支援模糊搜尋，輸入部分關鍵字即可！');
        }
      }
    }
    return ContentService.createTextOutput(JSON.stringify({ status: 'ok' }))
      .setMimeType(ContentService.MimeType.JSON);
  } catch (error) {
    Logger.log('doPost 錯誤：' + error.message);
    return ContentService.createTextOutput(JSON.stringify({ status: 'error' }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

逐行解釋：

  • JSON.parse(e.postData.contents) — 解析 Line 傳來的 JSON 資料
  • event.type === 'message' — 確認是訊息事件（不是加好友、退出群組等）
  • event.message.type === 'text' — 確認是文字訊息（不是圖片、貼圖等）
  • userMessage.indexOf('查') === 0 — 判斷訊息是否以「查」字開頭（位置 0 = 最前面）
  • userMessage.substring(1).trim() — 去掉第一個字（「查」或「找」），再去掉前後空白，取得純關鍵字
  • handleLineSearch(replyToken, keyword) — 呼叫搜尋處理函式，內部會呼叫 searchData() 並格式化回覆
  • try...catch — 防呆機制，即使程式出錯也不會讓 Webhook 中斷
💡 指令設計邏輯
  • 以「查」或「找」開頭 → 進入搜尋模式（例如「查 業務」→ 關鍵字 = 「業務」）
  • 傳「使用說明」或「幫助」→ 顯示操作指引
  • 其他訊息 → 目前不回覆（你可以自行加上回覆邏輯）

🔍 程式碼解析 4：前端 HTML（google.script.run）

這段的目的：前端「查詢介面.html」中的 JavaScript，透過 google.script.run 呼叫後端的 searchData() 函式，並把結果渲染成卡片。

// 📄 前端 查詢介面.html 中的 JavaScript

function doSearch() {
  var keyword = document.getElementById('keyword').value.trim();
  var resultsDiv = document.getElementById('results');

  if (!keyword) {
    resultsDiv.innerHTML = '<div class="no-result">請輸入搜尋關鍵字</div>';
    return;
  }

  // 顯示載入中
  resultsDiv.innerHTML = '<div class="loading">搜尋中</div>';

  // 🔥 核心：呼叫後端的 searchData 函式
  google.script.run
    .withSuccessHandler(showResults)   // 成功 → 呼叫 showResults()
    .withFailureHandler(showError)     // 失敗 → 呼叫 showError()
    .searchData(keyword);              // 呼叫後端的 searchData(keyword)
}

✅ withSuccessHandler

後端函式成功回傳時執行

  • 參數 = 後端函式的 return 值
  • 這裡就是 searchData() 回傳的陣列
  • showResults(data) 把陣列渲染成卡片

❌ withFailureHandler

後端函式發生錯誤時執行

  • 參數 = Error 物件
  • 用來顯示友善的錯誤訊息
  • showError(error) 顯示錯誤提示

結果渲染邏輯 showResults()

function showResults(data) {
  var resultsDiv = document.getElementById('results');
  var keyword = document.getElementById('keyword').value.trim();

  if (!data || data.length === 0) {
    resultsDiv.innerHTML = '🔍 找不到「' + keyword + '」的相關資料';
    return;
  }

  var html = '搜尋「' + keyword + '」找到 ' + data.length + ' 筆資料';
  for (var i = 0; i < data.length; i++) {
    var keys = Object.keys(data[i]);
    html += '<div class="result-card">';
    for (var j = 0; j < keys.length; j++) {
      html += keys[j] + '：' + data[i][keys[j]];  // 欄位名：欄位值
    }
    html += '</div>';
  }
  resultsDiv.innerHTML = html;
}
💡 關鍵觀念
google.script.run 是非同步的！呼叫後不會等結果，而是繼續往下跑。結果出來後才呼叫 Handler。這就是為什麼要搭配「載入動畫」— 讓使用者知道系統正在處理中。
